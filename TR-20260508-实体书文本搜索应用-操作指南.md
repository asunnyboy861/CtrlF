#### 2.1 相机预览管理器 (CameraManager.swift)

```swift
import AVFoundation
import SwiftUI

class CameraManager: NSObject, ObservableObject {
    let session = AVCaptureSession()
    @Published var isSessionRunning = false
    private var videoDataOutput: AVCaptureVideoDataOutput!
    private let videoDataOutputQueue = DispatchQueue(
        label: "com.ctrlf.videoDataOutputQueue",
        qos: .userInteractive
    )
    var onFrameCaptured: ((CVImageBuffer) -> Void)?

    override init() {
        super.init()
        configureSession()
    }

    func configureSession() {
        session.beginConfiguration()
        session.sessionPreset = .photo

        guard let videoDevice = AVCaptureDevice.default(
            .builtInWideAngleCamera,
            for: .video,
            position: .back
        ) else {
            session.commitConfiguration()
            return
        }

        guard let videoDeviceInput = try? AVCaptureDeviceInput(device: videoDevice),
              session.canAddInput(videoDeviceInput) else {
            session.commitConfiguration()
            return
        }
        session.addInput(videoDeviceInput)

        videoDataOutput = AVCaptureVideoDataOutput()
        videoDataOutput.alwaysDiscardsLateVideoFrames = true
        videoDataOutput.setSampleBufferDelegate(self, queue: videoDataOutputQueue)

        if session.canAddOutput(videoDataOutput) {
            session.addOutput(videoDataOutput)
        }

        session.commitConfiguration()
    }

    func startSession() {
        DispatchQueue.global(qos: .userInitiated).async { [weak self] in
            self?.session.startRunning()
            DispatchQueue.main.async {
                self?.isSessionRunning = true
            }
        }
    }

    func stopSession() {
        DispatchQueue.global(qos: .userInitiated).async { [weak self] in
            self?.session.stopRunning()
            DispatchQueue.main.async {
                self?.isSessionRunning = false
            }
        }
    }
}

extension CameraManager: AVCaptureVideoDataOutputSampleBufferDelegate {
    func captureOutput(
        _ output: AVCaptureOutput,
        didOutput sampleBuffer: CMSampleBuffer,
        from connection: AVCaptureConnection
    ) {
        guard let pixelBuffer = CMSampleBufferGetImageBuffer(sampleBuffer) else { return }
        onFrameCaptured?(pixelBuffer)
    }
}
```

#### 2.2 实时文字识别引擎 (TextRecognitionEngine.swift)

```swift
import Vision
import Combine

class TextRecognitionEngine: ObservableObject {
    @Published var recognizedTextItems: [RecognizedTextItem] = []
    @Published var matchedItems: [MatchedTextItem] = []
    @Published var isProcessing = false

    private var searchQuery = ""
    private var processingFrameCount = 0
    private let frameSkipInterval = 2

    func updateSearchQuery(_ query: String) {
        searchQuery = query.lowercased().trimmingCharacters(in: .whitespaces)
    }

    func processFrame(_ pixelBuffer: CVImageBuffer) {
        processingFrameCount += 1
        guard processingFrameCount % frameSkipInterval == 0 else { return }
        guard !searchQuery.isEmpty else {
            DispatchQueue.main.async { self.matchedItems = [] }
            return
        }

        let request = VNRecognizeTextRequest { [weak self] request, error in
            guard let observations = request.results as? [VNRecognizedTextObservation],
                  error == nil else { return }
            self?.handleRecognitionResults(observations, pixelBuffer: pixelBuffer)
        }

        request.recognitionLevel = .accurate
        request.recognitionLanguages = ["en-US", "en-GB"]
        request.usesLanguageCorrection = true
        request.minimumTextHeight = 0.02

        let handler = VNImageRequestHandler(cvPixelBuffer: pixelBuffer, options: [:])
        DispatchQueue.global(qos: .userInteractive).async {
            try? handler.perform([request])
        }
    }

    private func handleRecognitionResults(
        _ observations: [VNRecognizedTextObservation],
        pixelBuffer: CVImageBuffer
    ) {
        var matches: [MatchedTextItem] = []

        for observation in observations {
            guard let candidate = observation.topCandidates(1).first else { continue }
            let text = candidate.string.lowercased()
            let boundingBox = observation.boundingBox

            if text.contains(searchQuery) {
                let matchRange = text.range(of: searchQuery)
                let matchedText = MatchedTextItem(
                    fullText: candidate.string,
                    boundingBox: boundingBox,
                    confidence: candidate.confidence
                )
                matches.append(matchedText)
            }
        }

        DispatchQueue.main.async {
            self.matchedItems = matches
        }
    }
}

struct RecognizedTextItem {
    let text: String
    let boundingBox: CGRect
    let confidence: Float
}

struct MatchedTextItem {
    let fullText: String
    let boundingBox: CGRect
    let confidence: Float
}
```

#### 2.3 高亮叠加视图 (HighlightOverlayView.swift)

```swift
import SwiftUI

struct HighlightOverlayView: View {
    let matchedItems: [MatchedTextItem]
    let viewSize: CGSize

    var body: some View {
        Canvas { context, size in
            for item in matchedItems {
                let rect = convertBoundingBox(item.boundingBox, in: size)
                let highlightRect = roundRect(in: rect, radius: 4)

                context.fill(
                    highlightRect,
                    with: .color(.yellow.opacity(0.35))
                )
                context.stroke(
                    highlightRect,
                    with: .color(.yellow),
                    lineWidth: 2
                )
            }
        }
        .allowsHitTesting(false)
    }

    private func convertBoundingBox(_ box: CGRect, in size: CGSize) -> CGRect {
        let x = box.origin.x * size.width
        let y = (1 - box.origin.y - box.height) * size.height
        let width = box.width * size.width
        let height = box.height * size.height
        return CGRect(x: x, y: y, width: width, height: height)
    }

    private func roundRect(in rect: CGRect, radius: CGFloat) -> Path {
        Path(roundedRect: rect, cornerRadius: radius)
    }
}
```

#### 2.6 搜索历史管理 (SearchHistoryManager.swift)

```swift
import SwiftData
import Foundation

@Model
class SearchHistoryItem {
    var query: String
    var timestamp: Date
    var matchCount: Int

    init(query: String, matchCount: Int) {
        self.query = query
        self.timestamp = Date()
        self.matchCount = matchCount
    }
}

class SearchHistoryManager {
    private let maxHistoryCount = 50

    func addSearch(query: String, matchCount: Int, context: ModelContext) {
        let item = SearchHistoryItem(query: query, matchCount: matchCount)
        context.insert(item)
        trimHistory(context: context)
    }

    func getRecentSearches(context: ModelContext) -> [SearchHistoryItem] {
        let descriptor = FetchDescriptor<SearchHistoryItem>(
            sortBy: [SortDescriptor(\.timestamp, order: .reverse)]
        )
        return (try? context.fetch(descriptor)) ?? []
    }

    private func trimHistory(context: ModelContext) {
        let descriptor = FetchDescriptor<SearchHistoryItem>(
            sortBy: [SortDescriptor(\.timestamp, order: .reverse)]
        )
        guard let items = try? context.fetch(descriptor),
              items.count > maxHistoryCount else { return }
        for item in items.dropFirst(maxHistoryCount) {
            context.delete(item)
        }
    }
}
```

```swift
// Apple Vision 文字识别核心API
import Vision

// 创建文字识别请求
let request = VNRecognizeTextRequest { request, error in
    guard let observations = request.results as? [VNRecognizedTextObservation] else { return }
    for observation in observations {
        if let candidate = observation.topCandidates(1).first {
            let text = candidate.string
            let boundingBox = observation.boundingBox
            let confidence = candidate.confidence
        }
    }
}

// 配置识别参数
request.recognitionLevel = .accurate        // 精确模式 (vs .fast)
request.recognitionLanguages = ["en-US"]     // 识别语言
request.usesLanguageCorrection = true        // 语言纠错
request.minimumTextHeight = 0.02             // 最小文字高度比例

// 执行识别
let handler = VNImageRequestHandler(cvPixelBuffer: pixelBuffer)
try handler.perform([request])

// AVFoundation 相机核心API
import AVFoundation

let session = AVCaptureSession()
session.sessionPreset = .photo

let device = AVCaptureDevice.default(.builtInWideAngleCamera, for: .video, position: .back)
let input = try AVCaptureDeviceInput(device: device!)
session.addInput(input)

let output = AVCaptureVideoDataOutput()
output.setSampleBufferDelegate(self, queue: processingQueue)
session.addOutput(output)

// 手电筒控制
if device?.hasTorch == true {
    try device?.lockForConfiguration()
    device?.torchMode = .on  // or .off
    device?.unlockForConfiguration()
}
```
